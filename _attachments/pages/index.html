jQuery examples:

get objects in a bounding box:
              $.ajax({
                url: "http://civicapi.com/somedatabase/geojson",
                dataType: 'jsonp',
                data: {
                  "bbox":"lon,lat,lon,lat"// lower left, upper right
                },
                success: function(data){
                  // manipulate data
                }
              });
get available datasets:
              $.ajax({
                url: "http://civicapi.com/datasets",
                dataType: 'jsonp',
                success: function(data){
                  // manipulate data
                }
              });
get object metadata:
              $.ajax({
                url: "http://civicapi.com/somedatabase/somedocumentid",
                dataType: 'jsonp',
                success: function(data){
                  // manipulate data
                }
              });
